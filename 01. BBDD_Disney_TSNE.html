# Creando función para convertir texto en formato Markdown a DataFrame de pandas

def convertidor_mkd_df(md_text: str) -> pd.DataFrame:

    lines = [ln for ln in md_text.strip().splitlines() if '|' in ln]
    if not lines:
        return pd.DataFrame()

    def split_row(ln):
        parts = [c.strip() for c in ln.strip().strip('|').split('|')]
        return parts

    header = split_row(lines[0])
    idx = 1
    if re.fullmatch(r'\s*\|?\s*:?-{3,}:?\s*(\|\s*:?-{3,}:?\s*)+\|?\s*', lines[1]):
        idx = 2

    rows = [split_row(ln) for ln in lines[idx:] if ln.strip().strip('|').strip()]
    rows = [r + ['']*(len(header)-len(r)) if len(r) < len(header) else r[:len(header)] for r in rows]
    df = pd.DataFrame(rows, columns=[c.strip() for c in header])
    return df
